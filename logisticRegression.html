# Set colors to green and blue

h = 0.02
figure = plt.figure(figsize = (24, 12))
        
x_min, x_max = X[:, 0].min() - .5, X[:, 0].max() + .5
y_min, y_max = X[:, 1].min() - .5, X[:, 1].max() + .5

xx, yy = np.meshgrid(np.arange(x_min, x_max, h), np.arange(y_min, y_max, h))

# I decided to use the binary (grayscale) colourmap for the decision boundary, 
# since using the same colours as the datapoints makes it hard to see.
# To use a colormap of blue and green change the following line to
# cm = plt.cm.winter
cm = plt.cm.binary
cm_bright = ListedColormap(['#00FF00', '#0000FF'])

ax1 = plt.subplot(1, 2, 1, title="i) Plot of training data and decision boundary gradient")

# Plot training points on first plot
ax1.scatter(X_train[:, 0], X_train[:, 1], c=Y_train, cmap=cm_bright)

Z = model.decision_function(np.c_[xx.ravel(), yy.ravel()])
Z = Z.reshape(xx.shape)

# Plot decision boundary on first plot
ax1.contourf(xx, yy, Z, cmap = cm, alpha = 0.8)

# Identify misclassified points
misclassified = np.where(Y_test != model.predict(X_test))
colours = []
for x, y in zip(X_test, Y_test):
    x = [x]
    if y != model.predict(x):
        colours.append('red')
    elif y == 1:
        colours.append('blue')
    else:
        colours.append('green')

ax2 = plt.subplot(1, 2, 2, title="ii) Plot of testing data and decision boundary gradient with masclassified points in red")

# Plot training points on second plot
ax2.scatter(X_test[:, 0], X_test[:, 1], c=colours, cmap=cm_bright)

# Plot decision boundary on second plot
ax2.contourf(xx, yy, Z, cmap = cm, alpha = 0.8)
